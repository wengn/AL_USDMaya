391 template<typename T>
392 MStatus DgNodeTranslator::setVec3Anim(MObject node, MObject attr, const UsdGeomXformOp op, double conversionFactor)
393 {
394  MPlug plug(node, attr);
395  MStatus status;
396  const char* const xformErrorCreate = "DgNodeTranslator:setVec3Anim error creating animation curve";
397 
398  MFnAnimCurve acFnSetX;
399  acFnSetX.create(plug.child(0), NULL, &status);
400  AL_MAYA_CHECK_ERROR(status, xformErrorCreate);
401 
402  MFnAnimCurve acFnSetY;
403  acFnSetY.create(plug.child(1), NULL, &status);
404  AL_MAYA_CHECK_ERROR(status, xformErrorCreate);
405 
406  MFnAnimCurve acFnSetZ;
407  acFnSetZ.create(plug.child(2), NULL, &status);
408  AL_MAYA_CHECK_ERROR(status, xformErrorCreate);
409 
410  std::vector<double> times;
411  op.GetTimeSamples(&times);
412 
413  const char* const xformErrorKey = "DgNodeTranslator:setVec3Anim error setting key on animation curve";
414 
415  T value(0);
416  for(auto const& timeValue: times)
417  {
418  const bool retValue = op.GetAs<T>(&value, timeValue);
419  if (!retValue) continue;
420 
421  MTime tm(timeValue, MTime::kFilm);
422 
423  switch (acFnSetX.animCurveType())
424  {
425  case MFnAnimCurve::kAnimCurveTL: // time->distance: translation
426  case MFnAnimCurve::kAnimCurveTA: // time->angle: rotation
427  case MFnAnimCurve::kAnimCurveTU: // time->double: scale
428  {
429  acFnSetX.addKey(tm, value[0] * conversionFactor, MFnAnimCurve::kTangentGlobal, MFnAnimCurve::kTangentGlobal, NULL, &status);
430  AL_MAYA_CHECK_ERROR(status, xformErrorKey);
431  acFnSetY.addKey(tm, value[1] * conversionFactor, MFnAnimCurve::kTangentGlobal, MFnAnimCurve::kTangentGlobal, NULL, &status);
432  AL_MAYA_CHECK_ERROR(status, xformErrorKey);
433  acFnSetZ.addKey(tm, value[2] * conversionFactor, MFnAnimCurve::kTangentGlobal, MFnAnimCurve::kTangentGlobal, NULL, &status);
434  AL_MAYA_CHECK_ERROR(status, xformErrorKey);
435  break;
436  }
437  default:
438  {
439  break;
440  }
441  }
442  }
443 
444  return MS::kSuccess;
445 }
