58 template<typename AFnPlugin>
59 MStatus registerPlugin(AFnPlugin& plugin)
60 {
61  GlfGlewInit();
62 
63  if(!MGlobal::optionVarExists("AL_usdmaya_selectMode"))
64  {
65  MGlobal::setOptionVarValue("AL_usdmaya_selectMode", 0);
66  }
67 
68  if(!MGlobal::optionVarExists("AL_usdmaya_selectResolution"))
69  {
70  MGlobal::setOptionVarValue("AL_usdmaya_selectResolution", 10);
71  }
72 
73  MStatus status;
74 
75  // gpuCachePluginMain used as an example.
76  if (MGlobal::kInteractive == MGlobal::mayaState()) {
77  const auto otherPriority = MSelectionMask::getSelectionTypePriority("polymesh");
78  if (!MSelectionMask::registerSelectionType(AL::usdmaya::nodes::ProxyShape::s_selectionMaskName, otherPriority)) {
79  status.perror("Error registering selection mask!");
80  return MS::kFailure;
81  }
82 
83  MString cmd = "addSelectTypeItem(\"Surface\",\"";
84  cmd += AL::usdmaya::nodes::ProxyShape::s_selectionMaskName;
85  cmd += "\",\"";
86  cmd += "AL Proxy Shape";
87  cmd += "\")";
88 
89  status = MGlobal::executeCommand(cmd);
90  if (!status) {
91  status.perror("Error adding al_ProxyShape selection type!");
92  return status;
93  }
94  }

135  // Since AL_MAYA_DECLARE_NODE / AL_MAYA_DEFINE_NODE declare/define "creator"
136  // method, and AL_REGISTER_DEPEND_NODE registers "creator", in order to
137  // define custom creator, need to either 'override' one of those... chose to

141  MStatus status = plugin.registerNode(

143  AL::usdmaya::nodes::LayerManager::kTypeId,
144  AL::usdmaya::nodes::LayerManager::conditionalCreator,
145  AL::usdmaya::nodes::LayerManager::initialise);
146  if(!status) {
147  status.perror("unable to register depend node LayerManager");
148  return status;
149  }
150  }
151 
152  // generate the menu GUI + option boxes
153  AL::usdmaya::cmds::constructLayerCommandGuis();
154  AL::usdmaya::cmds::constructProxyShapeCommandGuis();
155  AL::usdmaya::cmds::constructDebugCommandGuis();
156 
157  CHECK_MSTATUS(AL::maya::utils::MenuBuilder::generatePluginUI(plugin, "AL_usdmaya"));
158  AL::usdmaya::Global::onPluginLoad();
159  return status;
160 }

172  MStatus status;
173 
174  // gpuCachePluginMain used as an example.
175  if (MGlobal::kInteractive == MGlobal::mayaState()) {
176  MString cmd = "deleteSelectTypeItem(\"Surface\",\"";
177  cmd += AL::usdmaya::nodes::ProxyShape::s_selectionMaskName;
178  cmd += "\")";
179 
180  status = MGlobal::executeCommand(cmd);
181  if (!status) {
182  status.perror("Error removing al_ProxyShape selection type!");
183  return status;
184  }
185 
186  if (!MSelectionMask::deregisterSelectionType(AL::usdmaya::nodes::ProxyShape::s_selectionMaskName)) {
187  status.perror("Error deregistering selection mask!");
188  return MS::kFailure;
189  }
190  }
